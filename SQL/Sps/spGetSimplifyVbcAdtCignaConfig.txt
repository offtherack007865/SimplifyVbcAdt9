IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND OBJECT_ID = OBJECT_ID('dbo.spGetSimplifyVbcAdtCignaConfig'))
   drop proc [dbo].[spGetSimplifyVbcAdtCignaConfig]
GO
CREATE PROCEDURE [dbo].[spGetSimplifyVbcAdtCignaConfig] 
AS
BEGIN
SELECT 
	[ReadDirectory]
	,[InputFilenameContainsString]
	,[StartLineNumber]
	,[StartColumnNumber]
	,[InputFileArchiveDirectory]
	,[OutputFileArchiveDirectory]
	,[ToSimpleFileMoverReadDirectory]
	,[SimplifyVbcAdtBaseWebApiUrl]
	,[ExcelTemplateFullFilename]
	,[ImportArchiveFolder]
	,[ps_odsConnectionString]
	,[BulkInsertConnectionString]
	,[BulkInsertDbName]
	,[BulkInsertDbTableName]
	,[BulkInsertBaseWebApiUrl]
	,[EmailBaseWebApiUrl]
	,[EmailFromAddress]
	   ,STUFF((SELECT  '; ' + [EmailAddress]
         FROM    
         ( SELECT DISTINCT [EmailAddress]
             FROM  [dbo].[SimplifyVbcAdtCignaConfigEmailAddress] eml
            where eml.[SimplifyVbcAdtCignaConfigPk] = cfg.[Pk]
         ) x
         FOR
         XML PATH('')
         ), 1, 2, '') [Emailees]
	   
  FROM [dbo].[SimplifyVbcAdtCignaConfig] cfg
END
