IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND OBJECT_ID = OBJECT_ID('dbo.spGetHumanaCountForGivenFileDateFromOds'))
   drop proc [dbo].[spGetHumanaCountForGivenFileDateFromOds]
GO
CREATE PROCEDURE [dbo].[spGetHumanaCountForGivenFileDateFromOds] 
(
  @inputFileDate [datetime]
)
AS
BEGIN
SELECT 
  count(*) as [HumanaCount]
   FROM [dbo].[ADT_Humana] cfg
  where [FileDate] = @inputFileDate

END
