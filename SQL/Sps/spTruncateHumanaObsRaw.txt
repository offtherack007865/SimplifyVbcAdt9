IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND OBJECT_ID = OBJECT_ID('dbo.spTruncateHumanaObsRaw'))
   drop proc [dbo].[spTruncateHumanaObsRaw]
GO
CREATE PROCEDURE [dbo].[spTruncateHumanaObsRaw] 
AS
BEGIN

  truncate table [dbo].[HumanaObsRaw];
  
  declare @HumanaRawCount [int] = 0  

  select @HumanaRawCount = count(*)
    from [dbo].[HumanaObsRaw]
 
   declare @IsOk [bit]
   
   if (@HumanaRawCount = 0) begin
     set @IsOk = 1
   end
   else begin
     set @IsOk = 0
   end 
   
   select @IsOk [IsOk]
   
END
