IF EXISTS (SELECT * FROM sys.objects WHERE type = 'P' AND OBJECT_ID = OBJECT_ID('dbo.spGetCignaCountForGivenFileDateFromOds'))
   drop proc [dbo].[spGetCignaCountForGivenFileDateFromOds]
GO
CREATE PROCEDURE [dbo].[spGetCignaCountForGivenFileDateFromOds] 
(
  @inputFileDate [datetime]
)
AS
BEGIN
SELECT 
  count(*) as [CignaCount]
  FROM [dbo].[ADT_Cigna] cfg
  where [FileDate] = @inputFileDate
  
END
